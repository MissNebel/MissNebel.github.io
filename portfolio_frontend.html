                                import { Settings } from './../../providers/settings/settings';
                                import { Component } from '@angular/core';
                                import { NavController } from 'ionic-angular';
                                
                                @Component({
                                    selector: 'settings-page',
                                    templateUrl: 'settings-page.html'
                                })
                                export class SettingsPage {
                                    selectedTheme: String = "default-theme";
                                
                                    constructor(
                                        public navCtrl: NavController, 
                                        public settings: Settings) {
                                        this.settings.getActiveTheme().subscribe(themeName => this.selectedTheme = themeName);
                                        this.sections = "ChangeTheme", "ChangeAvatar";
                                    }
                                
                                    toggleAppTheme(themeName) {
                                        if (themeName === 'default-theme') {
                                            this.settings.setActiveTheme('default-theme');
                                        } else if(themeName === 'darkone-theme') {
                                            this.settings.setActiveTheme('darkone-theme');
                                        } else if(themeName === 'pinkone-theme') {
                                            this.settings.setActiveTheme('pinkone-theme');
                                        }
                                    }
                                }
                                
                                
                                
                                import { Injectable } from '@angular/core';
                                import { BehaviorSubject } from 'rxjs/Rx';
                                
                                @Injectable()
                                export class Settings  {
                                
                                public theme: BehaviorSubject<String>
                                
                                constructor() {
                                    this.theme = new BehaviorSubject('default-theme');
                                }
                                
                                setActiveTheme(themeName) {
                                    this.theme.next(themeName);
                                }
                                
                                getActiveTheme() {
                                    return this.theme.asObservable();
                                }
                                }
